<script>

  // Amenity toggle behaviour

  ['carpark','concierge','gym','cinema'].forEach(id => {

    const toggle = document.getElementById('toggle-' + id);

    const checkbox = document.getElementById(id);

    toggle.addEventListener('click', function(e) {

      e.preventDefault();

      checkbox.checked = !checkbox.checked;

      toggle.classList.toggle('active', checkbox.checked);

    });

  });


  async function calculate() {

    // Validate inputs

    const location = document.getElementById('location').value;

    const sqft = parseFloat(document.getElementById('sqft').value);

    const bedrooms = document.getElementById('bedrooms').value;

    const charge = parseFloat(document.getElementById('charge').value);


    if (!location || !sqft || !bedrooms || !charge) {

      showError('Please fill in all fields before calculating.');

      return;

    }


    if (sqft < 100 || sqft > 5000) {

      showError('Please enter a realistic flat size between 100 and 5,000 sq ft.');

      return;

    }


    if (charge < 100 || charge > 20000) {

      showError('Please enter a realistic annual service charge amount.');

      return;

    }


    // Build payload

    const payload = {

      location: location,

      sq_ft: sqft,

      bedrooms: bedrooms,

      annual_charge: charge,

      car_park: document.getElementById('carpark').checked ? 'Yes' : 'No',

      concierge: document.getElementById('concierge').checked ? 'Yes' : 'No',

      gym: document.getElementById('gym').checked ? 'Yes' : 'No',

      cinema: document.getElementById('cinema').checked ? 'Yes' : 'No'

    };


    // Show loading

    document.getElementById('loading').classList.add('visible');

    document.getElementById('error').classList.remove('visible');

    document.getElementById('result').classList.remove('visible');

    document.getElementById('submitBtn').disabled = true;


    try {

      const response = await fetch(' https://hook.eu1.make.com/p9mrqp1bcbg4t0g76fhzxq8qb9mp8tdi', {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify(payload)

      });


      const data = await response.json();


      if (data.status === 'success') {

        showResult(data, charge, sqft);

      } else {

        showError('We could not find a benchmark for your building type. Please contact us directly for a manual assessment.');

      }


    } catch (err) {

      showError('Something went wrong connecting to our database. Please try again in a moment.');

    } finally {

      document.getElementById('loading').classList.remove('visible');

      document.getElementById('submitBtn').disabled = false;

    }

  }


  function showResult(data, charge, sqft) {

    const pct = parseFloat(data.overpayment_percent);

    const userCost = parseFloat(data.user_cost_per_sqft);

    const benchMid = parseFloat(data.benchmark_mid);

    const benchLow = parseFloat(data.benchmark_low);

    const benchHigh = parseFloat(data.benchmark_high);


    const numberEl = document.getElementById('resultNumber');

    const descEl = document.getElementById('resultDesc');


    if (pct > 5) {

      numberEl.textContent = '+' + Math.round(pct) + '% above benchmark';

      numberEl.className = 'result-number over';

      descEl.textContent = 'Your service charge appears to be above what we would expect for a building like yours. This suggests there may be costs worth challenging. A full audit will tell you exactly where the excess lies.';

    } else if (pct < -5) {

      numberEl.textContent = Math.round(pct) + '% below benchmark';

      numberEl.className = 'result-number under';

      descEl.textContent = 'Your service charge appears to be within or below the expected range for a building like yours. That said, a charge being low does not mean it is free of unjustified costs - a full audit will give you complete certainty.';

    } else {

      numberEl.textContent = 'Within benchmark range';

      numberEl.className = 'result-number fair';

      descEl.textContent = 'Your service charge appears to be broadly in line with what we would expect for a building like yours. A full audit will confirm whether all charges are properly evidenced and recoverable.';

    }


    document.getElementById('userCost').textContent = '£' + userCost.toFixed(2) + '/sq ft';

    document.getElementById('benchmarkCost').textContent = '£' + benchMid.toFixed(2) + '/sq ft';

    document.getElementById('benchmarkRange').textContent = '£' + benchLow.toFixed(2) + ' – £' + benchHigh.toFixed(2);

    document.getElementById('annualCharge').textContent = '£' + charge.toLocaleString();


    document.getElementById('result').classList.add('visible');

    document.getElementById('result').scrollIntoView({ behavior: 'smooth', block: 'start' });

  }


  function showError(msg) {

    const el = document.getElementById('error');

    el.textContent = msg;

    el.classList.add('visible');

  }
